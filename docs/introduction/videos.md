<script type="text/javascript">
    function iFrameHeight() {
        var iframes= document.getElementsByClassName("iframe"); 
        let height = -1
        for(var i=0;i<iframes.length;i++) {
            ifm = iframes[i]
            if (height==-1){
                height = ifm.parentElement.clientWidth * 0.6
            }
            ifm.height = height + 'px'
        }
    } 
</script>
